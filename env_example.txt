# ===========================
# 5) Память и Сеть
# ===========================
# Порог для estimate_tokens(messages), после которого агент начинает рассматривать
# суммаризацию истории.
# Как считается estimate_tokens:
# - если доступен tiktoken: используется cl100k_base + небольшой overhead на сообщение;
# - если tiktoken недоступен: fallback-оценка len(text)//3 по содержимому сообщений и tool_calls.
# Это внутренний эвристический счётчик агента, а не точный provider-side token usage из статус-бара.
# К estimate_tokens добавляется SUMMARY_RESERVED_TOKENS: запас на системные инструкции,
# tool schemas и служебную provider-обвязку, которые провайдер считает в input tokens,
# но которых нет в сохранённой истории сообщений. Поставьте 0, если нужен только
# "чистый" счётчик истории.
# Дополнительно есть "мягкая зона": чуть выше порога суммаризация может быть отложена,
# если сжимать пока почти нечего или в истории слишком мало human-turns.
SESSION_SIZE=8000
SUMMARY_RESERVED_TOKENS=3000

# Сколько последних сообщений оставлять "сырыми" (без сжатия) при суммаризации.
# Для рабочих чатов обычно хватает 4-6.
SUMMARY_KEEP_LAST=4

# Настройки повторных попыток для сетевых запросов (Search, Fetch).
MAX_RETRIES=3
RETRY_DELAY=2
